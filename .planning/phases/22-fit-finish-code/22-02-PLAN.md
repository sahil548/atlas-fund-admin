**Step 3 — Update `src/app/api/documents/route.ts` POST handler to accept FormData and validate via Zod:**

```typescript
import { DocumentFormDataSchema } from "@/lib/schemas";

// Inside the POST handler:
const contentType = req.headers.get("content-type") ?? "";
if (contentType.includes("multipart/form-data")) {
  const form = await req.formData();

  // 1. Validate the file part separately — Zod doesn't natively model File objects.
  const file = form.get("file");
  if (!(file instanceof File) || file.size === 0) {
    return NextResponse.json({ error: "A file is required" }, { status: 400 });
  }

  // 2. Extract scalar fields and validate them via DocumentFormDataSchema.
  //    NOTE: parseBody() is for JSON bodies and does NOT support multipart — we
  //    call .safeParse() directly on the constructed plain object.
  const parsed = DocumentFormDataSchema.safeParse({
    name: form.get("name")?.toString() ?? "",
    category: form.get("category")?.toString() ?? "",
    firmId: form.get("firmId")?.toString() ?? "",
    associatedDealId: form.get("associatedDealId")?.toString() || undefined,
    associatedEntityId: form.get("associatedEntityId")?.toString() || undefined,
    associatedAssetId: form.get("associatedAssetId")?.toString() || undefined,
  });
  if (!parsed.success) {
    return NextResponse.json(
      { error: parsed.error.issues[0]?.message ?? "Invalid form data" },
      { status: 400 },
    );
  }
  const { name, category, firmId, associatedDealId, associatedEntityId, associatedAssetId } = parsed.data;

  // 3. Persist file. If the codebase already has a storage helper (e.g. putObject),
  //    use it. Otherwise, store to local filesystem or the same location the
  //    legacy JSON path uses — preserve whatever convention already exists.
  //    The key output: set document.fileUrl to the persisted URL.
  //    (Placeholder: in dev, write to `public/uploads/` and set fileUrl = `/uploads/${filename}`.)
  const buffer = Buffer.from(await file.arrayBuffer());
  const ext = file.name.split(".").pop() ?? "bin";
  const storedName = `${crypto.randomUUID()}.${ext}`;
  // ...call existing storage helper OR write to public/uploads/...
  const fileUrl = `/uploads/${storedName}`;

  const doc = await prisma.document.create({
    data: {
      firmId, name, category, fileUrl,
      size: file.size,
      mimeType: file.type || "application/octet-stream",
      associatedDealId: associatedDealId || null,
      associatedEntityId: associatedEntityId || null,
      associatedAssetId: associatedAssetId || null,
    },
  });
  return NextResponse.json(doc, { status: 201 });
}
// Fall through to the existing JSON handler (keep it so older callers still work).
```
