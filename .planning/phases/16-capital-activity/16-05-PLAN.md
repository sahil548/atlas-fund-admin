POST handler: Create transaction and auto-recalculate IRR/MOIC:
```typescript
// Parse body with CreateAssetTransactionSchema
// If type === "income": create IncomeEvent
// If type === "expense": create AssetExpense
// Then: recalculate asset IRR and MOIC

// IRR recalculation:
// 1. Get all income events for asset (positive cash flows)
// 2. Get cost basis as initial outflow (negative, at entry date)
// 3. Add current fair value as terminal value (positive, today)
// 4. Call xirr(cashFlows)
// 5. MOIC = (fairValue + totalIncome - totalExpenses) / costBasis
// 6. Update asset: prisma.asset.update({ where: { id }, data: { irr, moic } })

import { xirr } from "@/lib/computations/irr";
import { computeMOIC } from "@/lib/computations/metrics";

async function recalculateAssetMetrics(assetId: string) {
  const asset = await prisma.asset.findUnique({
    where: { id: assetId },
    include: { incomeEvents: true, expenses: true },
  });
  if (!asset) return;

  const cashFlows: { date: Date; amount: number }[] = [];

  // Initial investment (outflow)
  if (asset.entryDate) {
    cashFlows.push({ date: new Date(asset.entryDate), amount: -asset.costBasis });
  }

  // Income (inflows)
  for (const inc of asset.incomeEvents) {
    cashFlows.push({ date: new Date(inc.date), amount: inc.amount });
  }

  // Expenses (outflows)
  for (const exp of (asset as any).expenses || []) {
    cashFlows.push({ date: new Date(exp.date), amount: -exp.amount });
  }

  // Current value (terminal inflow)
  cashFlows.push({ date: new Date(), amount: asset.fairValue });

  const irr = xirr(cashFlows);
  const totalIncome = asset.incomeEvents.reduce((s, e) => s + e.amount, 0);
  const totalExpenses = ((asset as any).expenses || []).reduce((s: number, e: any) => s + e.amount, 0);
  const adjustedValue = asset.fairValue + totalIncome - totalExpenses;
  const moic = asset.costBasis > 0 ? adjustedValue / asset.costBasis : null;

  await prisma.asset.update({
    where: { id: assetId },
    data: { irr, moic },
  });
}
```

Return the created transaction + updated asset metrics in response.
