    In the CommandBarProvider (or in command-bar.tsx), use `usePathname()` to auto-detect page context:
    ```typescript
    // Auto-detect basic page context from URL pattern
    const pathname = usePathname();
    useEffect(() => {
      if (pathname.match(/^\/deals\/[^/]+$/)) {
        setPageContext({ pageType: "deal", entityId: pathname.split("/")[2] });
      } else if (pathname.match(/^\/assets\/[^/]+$/)) {
        setPageContext({ pageType: "asset", entityId: pathname.split("/")[2] });
      } else if (pathname.match(/^\/entities\/[^/]+$/)) {
        setPageContext({ pageType: "entity", entityId: pathname.split("/")[2] });
      } else if (pathname === "/dashboard") {
        setPageContext({ pageType: "dashboard" });
      } else {
        setPageContext({ pageType: "other" });
      }
    }, [pathname]);
    ```
