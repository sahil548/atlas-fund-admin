- Export the `buildTree` function so it can be tested:
```typescript
export function buildTree(entities: any[]) {
  const map = new Map(entities.map(e => [e.id, { ...e, childEntities: [] }]));
  const roots: any[] = [];
  for (const entity of map.values()) {
    if (entity.parentEntityId && map.has(entity.parentEntityId)) {
      map.get(entity.parentEntityId)!.childEntities.push(entity);
    } else {
      roots.push(entity);
    }
  }
  return roots;
}
```
