  ```typescript
  function handleDragEnd(event: DragEndEvent) {
    const { active, over } = event;
    if (!over || active.id === over.id) return;
    const oldIdx = localTasks.findIndex(t => t.id === active.id);
    const newIdx = localTasks.findIndex(t => t.id === over.id);
    const reordered = arrayMove(localTasks, oldIdx, newIdx);
    setLocalTasks(reordered);
    // Persist — only PATCH the moved tasks, not all
    reordered.forEach((task, idx) => {
      if (task.order !== idx) {
        fetch("/api/tasks", {
          method: "PATCH",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ id: task.id, order: idx }),
        });
      }
    });
  }
  ```
